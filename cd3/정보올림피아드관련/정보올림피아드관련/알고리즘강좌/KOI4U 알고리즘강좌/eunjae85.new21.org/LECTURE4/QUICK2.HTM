퀵 정렬 - 비재귀판

  재귀호출을 하지 않는 퀵 소트입니다.
 
  원래 임중근 님이 베이직으로 올린 소스를 C로 옮겨서
만들었다가 원래 갖고 있던 임중근 님의 소스를 잃어버리는 바람에
C 로 만든 그 소스를 다시 BASIC으로 옮겼습니다.
 
  그러다 보니 원작과는 좀 차이가 있을 겁니다.
몇 개 명령문과 변수는 제가 없애기도 하고 이름을 바꾸기도 했으니
양해를....
 
 
DEFINT A-Z
DECLARE SUB QuickSort (arr(), n)
DECLARE SUB show (arr(), n)
 
n = 10
DIM a(n)
 
RANDOMIZE TIMER
FOR i = 1 TO n
    a(i) = INT(RND * 100) + 1
NEXT i
 
CLS
PRINT : PRINT "Before sorting:"
show a(), n
 
QuickSort a(), n
 
PRINT : PRINT "After sorting:"
show a(), n
 
END
 
 
SUB QuickSort (arr(), n)
 
    DIM l

s(n), rs(n)
 
    v = 0
    ls(v) = 1
    rs(v) = n
    v = v + 1
 
    DO WHILE v > 0
        v = v - 1
        aa = ls(v)
        bb = rs(v)
        IF aa >= bb GOTO quicksort1
        center = arr(aa) / 2 + arr(bb) / 2
        i = aa
        j = bb
        DO
            WHILE (i < bb AND arr(i) < center): i = i + 1: WEND
            WHILE (j > aa AND arr(j) > center): j = j - 1: WEND
            IF i >= j THEN EXIT DO
            SWAP arr(i), arr(j)
            i = i + 1
            j = j - 1
        LOOP
        ls(v) = aa: rs(v) = i - 1: v = v + 1
        ls(v) = j + 1: rs(v) = bb: v = v + 1
quicksort1:
  LOOP
END SUB
 

SUB show (arr(), n)
    FOR i = 1 TO n
        PRINT arr(i);
    NEXT i
    PRINT
END SUB
