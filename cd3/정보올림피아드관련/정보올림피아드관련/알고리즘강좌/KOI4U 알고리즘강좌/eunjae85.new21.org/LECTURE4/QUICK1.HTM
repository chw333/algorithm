퀵 정렬 - 재귀판

하이텔 베이직 동호회에서 김문수(새파람)님의 자료를 허락을 맡고 갈무리 해 온 소스입니다. 원작자의 동의 없이는 다른 곳으로 옮길 수 없습니다.

DEFINT A-Z
DECLARE SUB QuickSort (arr(), l, r)
DECLARE SUB show (arr(), n)
 
n = 10
DIM a(n)
 
RANDOMIZE TIMER
FOR i = 1 TO n
    a(i) = INT(RND * 100) + 1
NEXT i
 
CLS
PRINT : PRINT "Before sorting:"
show a(), n
 
QuickSort a(), 1, n
 
PRINT : PRINT "After sorting:"
show a(), n
 
END
 
 
SUB QuickSort (arr(), first, last)
 
    v = arr((first + last) / 2)
    r = first
    l = last
 
    DO
        WHILE (arr(r) < v)
            r = r + 1
        WEND
 
        WHILE (arr(l) > v)
            l = l - 1
        WEND

 
        IF (r <= l) THEN
            SWAP arr(r), arr(l)
            r = r + 1
            l = l - 1
        END IF
    LOOP UNTIL (r > l)
 
    IF (first < l) THEN QuickSort arr(), first, l
    IF (r < last) THEN QuickSort arr(), r, last
 
END SUB
 
SUB show (arr(), n)
    FOR i = 1 TO n
        PRINT arr(i);
    NEXT i
    PRINT
END SUB
