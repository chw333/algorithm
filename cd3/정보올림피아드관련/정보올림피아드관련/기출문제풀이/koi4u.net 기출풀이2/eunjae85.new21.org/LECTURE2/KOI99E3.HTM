색종이 올려놓기

초등부에서 동적 계획법 문제가 나왔군요. 방법만 간단히 설명해 드리도록 하겠습니다.

일단 색종이들을 x좌표 크기순으로 소트를 합니다. 그러면 색종이의 y좌표들도 재배열이 되는데, 앞서 살펴본 적이 있던 up sequence문제와 똑같은 문제가 됩니다. y좌표들에 대한 최대 up sequence를 찾아서 출력해 주면 됩니다.

아래 소스는 박준혁님이 수고해 주셨습니다.

program paper_problem;

const
     infile             =       'input.txt';
     outfile            =       'output.txt';

var
   n                                    :       integer;
   paper_x, paper_y                     :       array[ 0..100 ] of integer;
   dyn                                  :       array[ 0..100 ] of integer;

procedure input;

var
   inf          :       text;
   i            :       integer;

begin
     assign( inf, infile );
     reset( inf );

     read( inf, n );

     for i := 1 to n do
         read( inf, paper_x[ i ], paper_y[ i ] );

     close( inf );
end;

procedure swap( var a, b : integer );

var
   p    :       integer;

begin
     p := a;
     a := b;
     b := p;
end;

procedure solve;

var
   i, j         :       integer;
   max, maxi    :       integer;

begin
     for i := 1 to n do
         if paper_x[ i ] < paper_y[ i ] then swap( paper_x[ i ], paper_y[ i ] );

     for i := 1 to n do
         for j := i + 1 to n do
         begin
              if paper_x[ i ] > paper_x[ j ] then
              begin
                   swap( paper_x[ i ], paper_x[ j ] );
                   swap( paper_y[ i ], paper_y[ j ] );
              end;
              if ( paper_x[ i ] = paper_x[ j ] ) and
                 ( paper_y[ i ] > paper_y[ j ] ) then
              begin
                   swap( paper_x[ i ], paper_x[ j ] );
                   swap( paper_y[ i ], paper_y[ j ] );
              end;

         end;

     dyn[ 0 ] := 0;

     paper_x[ 0 ] := 0;
     paper_y[ 0 ] := 0;

     for i := 1 to n do
     begin
          max := - maxint;

          for j := 0 to i - 1 do
          begin
               if ( paper_x[ i ] >= paper_x[ j ] ) and
                  ( paper_y[ i ] >= paper_y[ j ] ) and
                  ( max < dyn[ j ] ) then
                  begin
                       max := dyn[ j ];
                       maxi := j;
                  end;
          end;
          dyn[ i ] := max + 1;
     end;
end;

procedure output;

var
   ouf  :       text;
   i    :       integer;
   max  :       integer;

begin
     max := dyn[ 1 ];
     for i := 1 to n do
         if dyn[ i ] > max then max := dyn[ i ];

     assign( ouf, outfile );
     rewrite( ouf );

     writeln( ouf, max );

     close( ouf );
end;

begin
     input;
     solve;
     output;
end.
