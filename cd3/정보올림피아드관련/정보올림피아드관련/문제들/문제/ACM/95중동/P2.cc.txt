/* marble.c */

#include <iostream.h>
#include <math.h>

float pi;

main ( int argc, char ** argv )
{
 float alphaDegrees, alpha, R, height;

 pi = 4*atan(1.0);

 while(1)
 {
  //  scanf("%f%f", &alphaDegrees, &R );

  cin >> alphaDegrees;
  cin >> R;

  if ( alphaDegrees == 0 ) break;

  alpha = alphaDegrees * pi / 180.0;

  height = R * (1.0 + 1.0/sin(alpha/2));

  //  printf("%f\n", height);

  cout << height << endl;
 }
}
