/* surface.cc */

#include <iostream.h>

#define MAXROWS 30
#define MAXCOLUMNS 25

int kase = 0;
char newLine;

int N;
char readChar;
char image[MAXROWS][MAXCOLUMNS];
int blanksPerRow[MAXROWS];
int minBlanksPerRow;
int totalBlanks;


int main ()
{
 int row, column;

 while(1)
 {
  totalBlanks = 0;

  cin >> N;
  if ( N == 0 ) break;

  kase++;

  minBlanksPerRow = MAXCOLUMNS;
  for(row=0;row<N;row++)
  {
   blanksPerRow[row] = 0;
   cin.get(readChar);
   for(column=0;column<MAXCOLUMNS;column++)
   {
    cin.get(readChar);
    
    image[row][column] = readChar;
    if ( image[row][column] == ' ' ) blanksPerRow[row]++;
   }

  totalBlanks += blanksPerRow[row];
  if ( blanksPerRow[row] < minBlanksPerRow )
    minBlanksPerRow = blanksPerRow[row];
  }
  cout << totalBlanks - N * minBlanksPerRow << endl;
  // printf("%d\n", totalBlanks - N * minBlanksPerRow );
 }
 return 0;
}
