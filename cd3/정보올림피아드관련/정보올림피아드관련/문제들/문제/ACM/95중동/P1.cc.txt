#include <iostream.h>

char buf[100];
int dividend;
int divisor = 13;
int quotientDigit;
int length;

int main()
{
 int i;
 int firstDigit;
 while ( 1 )
 {
  cin >> buf;
  if (cin.eof()) break;

  length = strlen ( buf );
  firstDigit = 1;
  dividend = 0;
  for(i=0; i<length; i++)
  {
   dividend = 10 * dividend + (buf[i] -'0');
   quotientDigit = dividend / divisor;
   dividend = dividend - quotientDigit * divisor;
   if ( ! firstDigit )
    cout << quotientDigit;
   else if ( quotientDigit != 0 )
   {
    firstDigit = 0;
    cout << quotientDigit;
   }
   else ; /* it's a zero leading digit */
  }
  if ( firstDigit )
    cout << 0;
  cout << endl;
 }
 return 0;
}
