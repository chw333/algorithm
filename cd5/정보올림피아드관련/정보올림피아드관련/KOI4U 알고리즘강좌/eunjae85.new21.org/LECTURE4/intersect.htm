두 선분이 있을 때 서로 교차하는지 교차하지 않는지를 검사하는 것은 육안으로는 매우 쉽게 알 수 있지만 컴퓨터가 계산하기에는 그리 쉽지 않습니다. 하지만 앞서 배운 선분의 방향성 검사 함수를 이용하면 두 선분 a-b와 c-d가 교차하는지를 쉽게 판별할 수 있습니다.

다만 주의할 것은 아래의 intersect함수는 완전교차(한 점이 다른 선분 위에 있다던가 하지 않는 교차)의 경우에만 제대로 동작한다는 것입니다. 조금만 수정하면 완전교차가 아닌 경우도 검출할 수 있습니다.

type
  Tpoint = record
    x, y : longint;
  end;

function left (a, b, c : Tpoint) : boolean;
var
  t : longint;
begin
  t := a.x * b.y + b.x * c.y + c.x * a.y
       - a.x * c.y - b.x * a.y - c.x * b.y;

  if t > 0 then left := true
           else left := false;
end;

function intersect (a, b, c, d : Tpoint) : boolean;
begin
  intersect := (left (a, b, c) xor left (a, b, d)) and
               (left (c, d, a) xor left (c, d, b));
end;

var
  a, b, c, d : Tpoint;
begin
  a.x := 0;
  a.y := 0;
  b.x := 5;
  b.y := 5;
  c.x := 5;
  c.y := 0;
  d.x := 0;
  d.y := 5;

  writeln (intersect (a, b, c, d));
end.
