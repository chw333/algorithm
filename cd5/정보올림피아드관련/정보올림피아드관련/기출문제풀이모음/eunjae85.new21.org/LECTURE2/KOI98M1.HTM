안정된 집단

그리 어려운 문제는 아닙니다. (아마 점수주기 문제였을 듯..) 하지만 약간 문제가 모호하고 명확하지 않아 보이므로 개념정리를 잘 하지 않고 프로그램 작성을 시작한다면 '어 안되네? 시불.. 처음부터 다시!'하는 결과가 생길지도 모릅니다 :)

일단 안정된 집단의 정의부터 제대로 내려야 합니다. 문제의 장황한 설명을 요약하면,

안정된 집단의 조건

1. 같은 그룹에 속한 사람들끼리는 서로 좋아해야 한다.

2. 다른 그룹에 속한 사람들끼리는 서로 싫어해야 한다.

3. 한 그룹은 최소한 2명 이상이다.

이렇게 세가지로 요약할 수 있습니다. 그리고 http://www.icc.or.kr 의 문제에서는 나와있지 않지만 입력데이터는 A가 B를 좋아하면 B도 A를 좋아하고, 싫어하는 경우도 마찬가지로 '대칭성'을 갖습니다.

이 문제는 특별히 알고리즘으로 설명하기 뭐하니까 그냥 주석을 붙여보도록 하겠습니다.

program KOI98M1;

var
  n             : integer; {사람 수}
  a             : array[1..100, 1..100] of integer; {좋아하는 관계를 나타냄}
  solution      : array[1..100] of integer; {각 사람이 어느 그룹에 속하는지}

  l             : integer;

procedure input_file;
var
  f     : text;
  i, j  : integer;
begin
  assign (f, 'input.txt');
  reset (f);

  read (f, n);

  for i := 1 to n do
    for j := 1 to n do
      read (f, a[i, j]);

  close (f);
end;

procedure solve;
var
  i, j, k       : integer;
  flag          : boolean;
begin
  solution[1] := 1; {1번 사람은 1번 그룹에 속한다고 하자}
  l := 1; {l은 현재 그룹의 개수}

  for i := 2 to n do begin
    for j := 1 to l + 1 do begin {i가 그룹 j에 들어갈 때를 검사}
      flag := false;
      for k := 1 to i - 1 do {i가 j에 속할 때 사람k와 문제가 있나?}
        if (j = solution[k]) xor (a[i, k] = 0) then flag := true;
{그룹이 같은데 싫어하거나 그룹이 다른데 좋아하는 경우가 있는지 체크}

      if not flag then begin {i번째 사람이 j그룹에 들어가도 아무 문제 없다면}
        solution[i] := j; {i번째 사람을 그룹j에 집어넣는다}
        if j = l + 1 then inc (l); {i번째 사람이 새로운 그룹일 경우 그룹수 + 1}
        break;
      end;
    end;
  end;
end;

procedure output_file;
var
  f             : text;
  i, j, c       : integer;
  flag          : boolean;
begin
  assign (f, 'output.txt');
  rewrite (f);

  for i := 1 to l do begin
    c := 0;
    for j := 1 to n do
      if solution[j] = i then inc (c);

    if c <= 1 then flag := true;
  end; {각 그룹에 속한 사람 수를 셈}

  if flag then begin
    writeln (f, 0); {어떤 그룹이 2명 미만이라면}
  end else begin
    writeln (f, l); {안정된 집단의 경우}

    for i := 1 to l do begin
      for j := 1 to n do
        if solution[j] = i then write (f, j, ' ');
      writeln (f);
    end;
  end;

  close (f);
end;

begin
  input_file;
  solve;
  output_file;
end.
