쉘 정렬

하이텔 베이직 동호회에서 김문수(새파람)님의 자료를 허락을 맡고 갈무리 해 온 소스입니다. 원작자의 동의 없이는 다른 곳으로 옮길 수 없습니다.

DEFINT A-Z
DECLARE SUB ShellSort (arr(), n)
DECLARE SUB show (arr(), n)
 
n = 10
DIM a(n)
 
RANDOMIZE TIMER
FOR i = 1 TO n
    a(i) = INT(RND * 100) + 1
NEXT i
 
CLS
PRINT : PRINT "Before sorting:"
show a(), n
 
ShellSort a(), n
 
PRINT : PRINT "After sorting:"
show a(), n
 
END
 
 
SUB ShellSort (arr(), n)
    gap = n
    WHILE gap <> 1
        gap = INT(gap / 2)
        DO
            done = 1
            FOR i = 1 TO n - gap
                IF arr(i) > arr(i + gap) THEN SWAP arr(i), arr(i + gap): done = 0
            NEXT i
        LOOP WHILE done = 0

    WEND
END SUB
 
 
SUB show (arr(), n)
    FOR i = 1 TO n
        PRINT arr(i);
    NEXT i
    PRINT
END SUB
