#include <iostream.h>

void main()

{

    int a[100];

    int i, count=0, sum=0;

    for(;;) {

        cin >> a[count];

        if( a[count]==0) break;

        count++;

    }

    for(i=0; i<count; i++) {

        sum+=a[i];

    }

    cout << sum;

}

 

 

설명

 

1. 몇 개를 받아야 할지 모르기 때문에 배열의 크기를 넉넉히 100정도로 잡았다.

3. for(){...}문이 끝나면, count변수에는 사람이 입력한 값이 a[]배열에 들어오고, 개수는

   count변수에 들어온다. 

4. a[]배열에 들어온 값을 count개수만큼 합산하여 출력한다.

▷소스코드

#include <iostream.h>

void main()

{

    int a[100];

    int i, count=0, 

       max, min, sum;

    for(;;) {

        cin >> a[count];

        if( a[count]==0) break;

        count++;

    }

    max=a[0];

    min=a[0];

    sum=a[0];

    for(i=1; i<count; i++) {

        if(max < a[i]) {

            max=a[i];

        }

        if(min > a[i]) {

            min=a[i];

        }

        sum=sum+a[i];

    }

    cout <<"가장 큰 수="<<max;

    cout <<"\n";

    cout <<"가장 작은 수="<<min;

cout <<"\n";

    cout <<"총합="<<sum<<"\n";

}

 

 

 

 

설명

 

1. 가장 큰 수, 가장 작은 수, 총합을 한꺼번에 처리하기 위해서 max, min, sum변수를 어떻게 사용했는지를 살펴보자.

2. 각각을 분리하여 생각하면 앞의 3,4,5번 문제와 동일함을 알 수 있다.

3. 여기에서 특이한 것은 총합을 구하는 부분이다. 가장 큰 수와 가장 작은 수를 함께 처리하기 위해서 sum의 출발을 sum=a[0]으로 하고, for문에서 i는 1부터 출발하여 a[i]의 원소가 a[1]부터 계산되도록 했다.

 

4. 만약에,

   for(i=0;;)으로 i는 0부터 출발한다면, 소스코드를 다음과 같이 고쳐야 할 것이다.

   max=-9999;

   min=9999;

   sum=0;

   for(i=0; i<count; i++) {

       if(max < a[i]) {

           max=a[i];

       }

       if(min > a[i]) {

                min=a[i];

       }

       sum=sum+a[i];

   }

5. max를 -9999로 설정해 주면, max와 a[0]값을 비교할 때 a[0]값이 더 크므로 max는 a[0]으로 설정되며, min을 9999로 설정해 주면 min과 a[0]을 비교하여 min은 a[0]으로 설정된다.
