◈ Queen 배치 문제
  • queen들이 서로 공격하는지를 검사하는 방법
      － col (i ): i 행에서 queen이 있는 열 위치
      － i 행과 k 행에서 queen이 같은 열에 위치하는 경우: col(i ) = col(k)
      － i 행과 k 행에서 queen이 대각선 상에 위치하는 경우: col(i ) - col(k ) = i - k 또는 col(i ) - col (k ) = k - i

      void queens ( index i )

        index j ;
        if ( promising ( i ))
          if ( i == n )
          cout << col [ i ] through col [ n ];
          else
            for ( j = 1; j < = n ; j ++) {
            col [ i + 1 ] = j ;
            queens ( i + 1 );
            }
      }
      bool promising ( index i )
      {
        index k ;
        bool switch ;
        k = 1;
        switch = true;
        while ( k < i && switch ) {
          if ( col [ i ] == col [ k ] || abs ( col [ i ] - col [ k ] ) == i - k )
          switch = false;
          k ++;
        }
        return switch;
      }
