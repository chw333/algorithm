DIM s AS INTEGER, e AS INTEGER

DIM kk AS INTEGER, x AS INTEGER

DIM I AS INTEGER, J AS INTEGER

CLS

x = 999

DO WHILE (x >= 5)

   INPUT " 숫자를 입력하시오 ... x< 5"; x

LOOP

kk = 2 * x + 1

DIM SHARED Buffer(kk, kk) AS INTEGER

FOR I = 1 TO x

   s = I: e = kk - I:  PutInbuffer s, e, 1

   s = I: e = kk - I:  PutInbuffer s, e, 0

   s = kk - I + 1: e = I + 1:  PutInbuffer s, e, 1

   s = kk - I + 1: e = I + 1:  PutInbuffer s, e, 0

NEXT I

Buffer(x + 1, x + 1) = kk * kk

FOR I = 1 TO kk

   FOR J = 1 TO kk

      PRINT USING "####"; Buffer(I, J);

   NEXT J

   PRINT : PRINT

NEXT I


SUB PutInbuffer (s AS INTEGER, e AS INTEGER, flag AS INTEGER)

   DIM J AS INTEGER

   DIM stepval AS INTEGER

  

   stepval = SGN(e - s)

   FOR J = s TO e STEP stepval

      cnt = cnt + 1

      IF flag = 1 THEN

         Buffer(J, s) = cnt

      ELSE

         Buffer(e + stepval, J) = cnt

      END IF

   NEXT J

END SUB
