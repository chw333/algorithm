#include <fstream.h>
#include <ctype.h>
#include <string.h>

int input( void );
void process( void );
void output( void );

int ta[ 10 ][ 10 ];
int where[ 10 ];
int n, m, kk = 1;

int main( void )
{
	while( kk )
	{
		if( ! input() ) { cout << "!!!dnuof ton elif" << endl; return 1; }
		process();
		output();
	}

	return 0;
}

int input( void )
{
	int l1, l2;

	ifstream in( "jaryo.dat" );

	if( ! in ) { in.close(); return 0; }
	for( l1 = 0; l1 < 10; l1++ )
	{
		for( l2 = 0; l2 < 10; l2++ )
		{
			in >> ta[ l1 ][ l2 ];
		}
	}

	in.close();

	cout << "Input column and row : ";
	cin >> n; cin.get(); cin >> m; cin.get();

	return 1;
}

void process( void )
{
	int l1, l2, l3;

	for( l1 = 0; l1 < n; l1++ )
	{
		where[ l1 ] = l1;
		for( l2 = l1 + 1; l2 < m; l2++ )
		{
			if( ta[ l1 ][ where[ l1 ] ] < ta[ l1 ][ l2 ] ) where[ l1 ] = l2;
		}
	}

	for( l1 = 1; l1 < n; l1++ )
	{
		for( l2 = l1; l2 > 0 && where[ l2 ] <= where[ l2 - 1 ]; l2-- )
		{
			where[ l2 - 1 ] = l2 - 1;
			for( l3 = l2; l3 < where[ l2 ]; l3++ )
			{
				if( ta[ l2 - 1 ][ where[ l2 - 1 ] ] < ta[ l2 - 1 ][ l3 ] ) where[ l2 - 1 ] = l3;
			}
		}
	}
}

void output( void )
{
	char str[ 100 ];
	int l, tmp1;
	int l1;

	for( l1 = 0; l1 < n; l1++ )
	{
		cout << "column " << l1 + 1 << "'s maximum is " << where[ l1 ] + 1 << " row, " << ta[ l1 ][ where[ l1 ] ] << endl;
	}
bb:
	cout << "Retry? "; cin.getline( str, 100, '\n' );
	l = strlen( str ); tmp1 = 0;
	for( l1 = 0; l1 < l; l1++ )
	{
		if( str[ l1 ] != ' ' ) tmp1++;
	}
	if( tmp1 != 1 ) goto bb;
	for( l1 = 0; l1 < l; l1++ )
	{
		if( str[ l1 ] != ' ' )
		{
			if( tolower( str[ l1 ] ) == 'y' ) kk = 1;
			else if( tolower( str[ l1 ] ) == 'n' ) kk = 0;
			else goto bb;
		}
	}
}
