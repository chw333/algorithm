#include <stdio.h>
FILE *outfile;
int size;
int x[10];
int ans[10], sw[10];
unsigned long sum;

void get_data(void)
{
	FILE *infile = fopen("input.txt", "r");
	fscanf(infile, "%d", &size);
	for(int i = 0; i < size; i++)
		fscanf(infile, "%d", &x[i]);
	fclose(infile);
}

void back(int cur)
{
	if (cur > size)
	{
		for(int i = 0; i < size; i++)
			fprintf(outfile, "%c", ans[i] + 48);
		fprintf(outfile, "\n");
		return;
	}
	int i;
	for(i = 0; i < size; i++)
	{
		if (!sw[i] && !((sum * 10 + x[i]) % cur) && (cur || x[i]))
		{
			sw[i] = 1;
			sum = sum * 10 + x[i];
			ans[cur - 1] = x[i];
			back(cur + 1);
			sw[i] = 0;
			sum = (sum - x[i]) / 10;
		}
	}
}

int main(void)
{
	outfile = fopen("output.txt", "w");
	get_data();
	back(1);
	fclose(outfile);
	return 0;
}
