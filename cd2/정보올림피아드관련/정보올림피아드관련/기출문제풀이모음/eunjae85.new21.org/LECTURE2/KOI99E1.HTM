다각형 그리기

단순한 문제입니다. '그리기 수열'을 회전시켜 주어서 일치하는지를 검사하면 됩니다. 반대방향으로도 같은 다각형이 그려질 수 있기 때문에 원래의 수열의 방향을 뒤집어서도 처리해 줍니다.

아래 소스는 박준혁님이 수고해 주셨습니다.

program koi_99_elementary_1;

const
     infile             =               'input.txt';
     outfile            =               'output.txt';
     reverse            :               array[ 1..4 ] of integer = ( 3, 4, 1, 2 );

type
    size                =       array[ 1..100 ] of integer;

var
   m, n                                 :       integer;
   standard                             :       size;
   process                              :       array[ 1..100 ] of size;
   outn                                 :       integer;
   out                                  :       array[ 1..100 ] of boolean;

procedure input;

var
   inf          :       text;
   i, j         :       integer;

begin
     assign( inf, infile );
     reset( inf );

     read( inf, n );

     for i := 1 to n do
         read( inf, standard[ i ] );

     read( inf, m );

     for i := 1 to m do
          for j := 1 to n do
              read( inf, process[ i ][ j ] );

     close( inf );
end;

procedure swap( var a, b : integer );

var
   p    :       integer;

begin
     p := a;
     a := b;
     b := p;
end;

function ok( a : size ) : boolean;

var
   k, i, j         :       integer;
   check           :       boolean;

begin

     ok := false;
     for k := 1 to 2 do
     begin

          for i := 1 to n do
              if a[ i ] = standard[ 1 ] then
              begin
                   check := true;
                   for j := 1 to n do
                       if a[ ( i + j - 2 ) mod n + 1 ] <> standard[ j ] then
                       begin
                            check := false;
                            break;
                       end;
                   if check then
                   begin
                        ok := true;
                        exit;
                   end;
              end;

          for i := 1 to n do
              a[ i ] := reverse[ a[ i ] ];

          for i := 1 to n div 2 do
              swap( a[ i ], a[ n - i + 1 ] );
     end;
end;

procedure solve;

var
   i    :       integer;

begin
     outn := 0;

     for i := 1 to m do
         out[ i ] := false;

     for i := 1 to m do
     begin
          if ok( process[ i ] ) then
          begin
               out[ i ] := true;
               inc( outn );
          end;
     end;
end;

procedure output;

var
   ouf          :       text;
   i, j         :       integer;

begin
     assign( ouf, outfile );
     rewrite( ouf );
     writeln( ouf, outn );
     for i := 1 to n do
         if out[ i ] then
         begin
              for j := 1 to n do
                  write( ouf, process[ i ][ j ], ' ' );
              writeln( ouf );
         end;

     close( ouf );
end;

begin
     input;
     solve;
     output;
end.
